    Sorted List (Custom Class):

    We use JavaScript’s built-in Array class to implement a SortedList. We use the sort method to sort the array.

    class SortedList {
  constructor() {
    this.list = [];
  }

  add(element) {
    this.list.push(element);
    this.list.sort((a, b) => a - b);
  }

  remove(element) {
    const index = this.list.indexOf(element);
    if (index !== -1) {
      this.list.splice(index, 1);
    }
  }

  get(index) {
    return this.list[index];
  }

  has(element) {
    return this.list.includes(element);
  }
}

const sortedList = new SortedList();
sortedList.add(3);
sortedList.add(1);
sortedList.add(2);

    Sorted Set (Custom Class):

    We use JavaScript’s built-in Set class to implement a SortedSet. We use a separate array to store the sorted elements.

    class SortedSet {
  constructor() {
    this.set = new Set();
    this.sortedArray = [];
  }

  add(element) {
    if (!this.set.has(element)) {
      this.set.add(element);
      this.sortedArray.push(element);
      this.sortedArray.sort((a, b) => a - b);
    }
  }

  remove(element) {
    if (this.set.has(element)) {
      this.set.delete(element);
      const index = this.sortedArray.indexOf(element);
      if (index !== -1) {
        this.sortedArray.splice(index, 1);
      }
    }
  }

  get(index) {
    return this.sortedArray[index];
  }

  has(element) {
    return this.set.has(element);
  }
}

const sortedSet = new SortedSet();
sortedSet.add(3);
sortedSet.add(1);
sortedSet.add(2);

    Linked List (Custom Class): 

    class Node {
  constructor(value) {
    this.value = value;
    this.next = null;
  }
}

class LinkedList {
  constructor() {
    this.head = null;
  }

  append(value) {
    const newNode = new Node(value);
    if (!this.head) {
      this.head = newNode;
    } else {
      let current = this.head;
      while (current.next) {
        current = current.next;
      }
      current.next = newNode;
    }
  }

  remove(value) {
    if (!this.head) return;

    if (this.head.value === value) {
      this.head = this.head.next;
    } else {
      let current = this.head;
      while (current.next) {
        if (current.next.value === value) {
          current.next = current.next.next;
          return;
        }
        current = current.next;
      }
    }
  }

  contains(value) {
    let current = this.head;
    while (current) {
      if (current.value === value) return true;
      current = current.next;
    }
    return false;
  }
}

const linkedList = new LinkedList();
linkedList.append(1);
linkedList.append(2);
linkedList.remove(1);
